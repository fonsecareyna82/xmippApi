889  const Matrix1D<double> &corner2)
890  {
891  Image<T> * Vol_aux;
892  std::vector<Image<T> * > LV_aux;

895  {

899  grid.universe2grid(corner1, grid.lowest);

901  grid.universe2grid(corner2, grid.highest);
902  grid.highest.selfCEIL();

911  STARTINGZ((*Vol_aux)()) = (int) ZZ(grid.lowest); // are stored as float

915  SPEED_UP_tempsInt;

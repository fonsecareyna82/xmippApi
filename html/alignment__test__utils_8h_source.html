40 static std::vector<Point2D<float>> generateShifts(const Dimensions &dims, size_t maxShift, std::mt19937 mt) { // pass copy as not to interfere with size generation
41  auto maxShiftSq = maxShift * maxShift;
42 
43  std::uniform_int_distribution<> dist(0, maxShift);
44  auto shifts = std::vector<Point2D<float>>();
45  shifts.reserve(dims.n());
46  for(size_t n = 0; n < dims.n(); ++n) {
47  // generate shifts so that the Euclidean distance is smaller than max shift
48  int shiftX = dist(mt);
49  int shiftXSq = shiftX * shiftX;
50  int maxShiftY = std::floor(sqrt(maxShiftSq - shiftXSq));
51  int shiftY = (0 == maxShiftY) ? 0 : dist(mt) % maxShiftY;
52  shifts.emplace_back(shiftX, shiftY);
53  }
54  return shifts;
55 }

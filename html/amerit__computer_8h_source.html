46  void check() const {
47  if (0 == hw.size()) {
48  REPORT_ERROR(ERR_VALUE_INCORRECT, "HW contains zero (0) devices");
49  }
50  if ( ! refDims.isValid()) {
51  REPORT_ERROR(ERR_LOGIC_ERROR, "'Ref' dimensions are invalid (contain 0)");
52  }
53  if ( ! otherDims.isValid()) {
54  REPORT_ERROR(ERR_LOGIC_ERROR, "'Other' dimensions are invalid (contain 0)");
55  }
56  if ( ! refDims.equalExceptNPadded(otherDims)) {
57  REPORT_ERROR(ERR_LOGIC_ERROR, "Dimensions of the reference and other signals differ");
58  }
59  if ((MeritType::OneToN == type)
60  && (1 != refDims.n())) {
61  REPORT_ERROR(ERR_LOGIC_ERROR, "More than one reference specified for type 1:N");
62  }
63  }
64 };

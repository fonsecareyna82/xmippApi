29 // struct is used for compatibility with OpenCL / C for Cuda
30 /* Instance of this struct represent a 2D array of arbitrary type.
31  * Data are being stored as a dynamic 2D array (i.e. not continuous block of memory).
32  * Access to elements will be the fastest if traversing in Y -> X order. */
33 template<typename T>
34 struct Array2D {
35 public:
36  /* Empty constructor */
37  Array2D() : xSize(0), ySize(0), data(NULL) {};
38  /* Constructor, allocates the data immediately */
39  Array2D(int xSize, int ySize) :
40  xSize(xSize), ySize(ySize) {
41  data = new T*[ySize];
42  for (int y = 0; y < ySize; y++) {
43  data[y] = new T[xSize];
44  for (int x = 0; x < xSize; x++) {
45  data[y][x] = (T) 0;
46  }
47  }
48  #if DEBUG
49  std::cout << "Array2D created (" << getXSize() << "x" << getYSize()
50  << ") at " << data << std::endl;
51  #endif
52  }
53 
54  ~Array2D() {
55  for (int y = 0; y < ySize; y++) {
56  delete[] data[y];
57  }
58  delete[] data;
59  #if DEBUG
60  std::cout << "Array2D deleted (" << getXSize() << "x" << getYSize()
61  << ") at " << data << std::endl;
62  #endif
63  data = 0;
64  }
65 
66  /* Method to access elements of the array */
67  T& operator()(int x, int y) const {
68  #if DEBUG
69  if (0 > x || x >= xSize)
70  std::cout << "Array2D " << data << " X=" << x << " out of range [0.." << getXSize() << ")\n";
71  if (0 > y || y >= ySize)
72  std::cout << "Array2D " << data << " Y=" << y << " out of range [0.." << getYSize() << ")\n";
73  std::cout << std::flush;
74  #endif
75  return data[y][x];
76  }
